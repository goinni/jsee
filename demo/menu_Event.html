part 2.获取键盘码 与 事件终止  
//简化书写
var $ = Element;
$(function(){
	//页面加载完成后执行以下内容
	$("#tempInput1").bind({
		onkeydown:function(){
			//获取键盘码
			var keycode = Event.keyCode();
			//只允许输入0~9数字
			if(keycode>=48 && keycode <=57){
				//提示按下的键盘码
				alert(Event.keyCode());
			}else{
				//终止事件响应
				Event.stop();
			}
		}
	});
});
