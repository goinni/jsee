part 1.截取字符串      
var text = "abc123456";
text.subString(3); // 结果为 123456
text.subString(3,6); //结果为 123
text.subString(3,6,true); //结果为 123...

part 2.去空格      
var text = " a b c 1 2 3 4 5 6 ";
text.trims(); // 结果为abc123456

part 3.检查A字符串是否在B字符串内      
var text_A  = "1a2346";
var text_B = "123456798abcdefg";
text_A.isBelong(text_B); //结果为true; [注]text_B缺省值为"A~Za~z0~9"

part 4.将html字符串转为节点对象 
var htmlStr = "<div>内容1</div><div>内容2</div>";
var newDom = htmlStr.toDom(); //转换后的节点对象都是newDom的子节点
var nDom = $(newDom).children();
nDom.eq(0).getHtml();//结果为: 内容1
nDom.eq(1).getHtml();//结果为: 内容2
