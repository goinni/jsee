part 1.定义一个Timer对象      
$(function(){
//页面加载完成已后执行以下内容
var pp = $("#tipbox").entity();
//定义一个Timer对象
var timer = new Timer(
	function(time){
    	//time为当前计时器运行的毫秒数
		if(time >= 5000) {
			//计时器启动5秒后停止,this表示当前timer对象
			this.stop();
		}
		//每隔100毫秒对数据进行累加一次
		pp.innerHTML =parseFloat(pp.innerHTML)+1;   
	}
,100);//每隔100毫秒执行一次
//启动计时器
timer.start();
});
