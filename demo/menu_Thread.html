part 1.定义一个Thread对象      
var i = 0;
var thread = new Thread(function(){
	var t = new Date().getSeconds();//获取当前秒数	
	if(t==59) this.stop(); //秒数为59秒是 停止线程; this表示当前线程
	if(t==20) this.sleep(3000); //秒数为20秒时暂停3秒后再执行
    //如果sleep()无参数则会一直等待下去，至到触发notify()方法时才会重新唤醒线程
	if(t==30) this.setTime(200);//秒数为30秒时重置线程执行频率为200毫秒。(初始值为1)
	//this.time = 1;//可以设置初始频率 单位毫秒
	document.getElementById("tipbox").innerHTML = "当前秒数为"+t +" 速度标记"+i++;
});
//启动线程,也可以new Thread(run).start();
thread.start();
