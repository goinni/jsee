part 1.定义一个Map对象        
var map = new Map();	//定义一个Map对象
    map.put('name','houzhenyu');	//向Map对象中存值
    map.put('age','27');
    map.put('sex','boy');

part 2. 循环一个Map对象   
//循环MAP对象
map.each(function(key,value){
	alert(key+"="+value);
})

part 3. 数据清空与删除   
map.remove(key) //从此映射中移除指定键的映射关系（如果存在）
map.clear() //从此映射中移除所有映射关系

part 4. 数据校验   
map.size() //返回此映射中的键-值映射关系数
map.isEmpty() //如果此映射不包含键-值映射关系，则返回 true

part 5. 从Map中取值   
map.get(key) //返回指定键所映射的值；如果对于该键来说，此映射不包含任何映射关系，则返回 undefined
如：part1中的map.get("age") 结果为27
